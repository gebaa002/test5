def classification_model_validation2(X,y):

    models = [KNeighborsClassifier(),LogisticRegression(),
    SGDClassifier(),SVC(),RandomForestClassifier(), tree.DecisionTreeClassifier(),
              AdaBoostClassifier(), GradientBoostingClassifier()]

    scores = []

    for model in models:

        cross = cross_val_score(model, X, y, cv=5).mean()
        scores.append(cross)

    d = {'Model': models, 'Score': scores}
    df_scores = pd.DataFrame(data=d)



    sns.barplot(x=df_scores['Model'],y=df_scores['Score'])
    plt.tick_params(axis='x', labelrotation = 90)
    print("Best score : ",models[scores.index(max(scores))])
    print("             ",max(scores))
